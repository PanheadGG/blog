/* 设计并测试一个矩形类（Rectangle），属性为矩形的左下与右上角的坐标，
矩形水平放置。操作为计算矩形周长与面积。测试包括用成员函数和普通函数。
解：这里的矩形的4边分别与x轴y轴平行，为最简单的情况。注意参数有缺省值
的函数的声明和定义格式。
*/

#include <iostream>
#include <cmath>
using namespace std;
class Rectangle 
{
    double left,bottom ;
    double right,top  ;
        public:
    Rectangle(double l=0, double b=0, double r=0, double t=0){
        left = l;
        bottom = b;
        right = r;
        top = t;
    }
    ~ Rectangle(){};  //析构函数，在此函数体为空
    void Assign(double l,double b,double r,double t){
        left = l;
        bottom = b;
        right = r;
        top = t;
    }
    double getLeft(){ return left;} // 以下四个函数皆为内联成员函数，返回各个点的坐标信息
    double getRight(){ return right;}
    double getTop(){return top;}
    double getBottom(){return bottom;}
    void Show(){
        cout<<"left-bottom point is ("<<left<<","<<bottom<<")\n";
        cout<<"right-top point is ("<<right<<","<<top<<")\n";
    }  //显示坐标信息
    double Area(){
        return abs((top-bottom)*(right-left));
    } //计算面积
    double Perimeter(){
        return (abs(top-bottom)+abs(right-left))*2;
    }  //计算周长
};



int main(){
	double x1,y1,x2,y2;
	cin>>x1>>y1>>x2>>y2;
	Rectangle rect;
	rect.Assign(x1,y1,x2,y2);
	rect.Show();
	cout<<"面积"<<rect.Area()<<'\t'<<"周长"<<rect.Perimeter()<<endl;
	return 0;
}

#include<iostream>
#include<cmath>
#define PI 3.14159265

using namespace std;

class Circle
{
    int radius;
    public:
    Circle(){}
    Circle(int r){
        radius = r;
    }
    void SetR(int r){
        radius = r;
    }
    int GetR(){
        return radius;
    }
    double GetCircumference(){
        return 2*PI*radius;
    }
    double GetAreaCircle(){
        return PI*radius*radius;
    }
};


int main(){
    int r;
    cin>>r;
	Circle cl1(r),cl2,cl3=cl1;
	cout<<"圆半径:"<<cl3.GetR()<<'\t'<<"圆周长:"<<cl3.GetCircumference()
		<<'\t'<<"圆面积:"<<cl3.GetAreaCircle()<<endl;
	cin>>r;
	cl2.SetR(r);
	cout<<"圆半径:"<<cl2.GetR()<<'\t'<<"圆周长:"<<cl2.GetCircumference()
		<<'\t'<<"圆面积:"<<cl2.GetAreaCircle()<<endl;
	return 0;
}

	void puttime(int ,int ,int );

//点击在此输入一行或多行代码

#include<iostream>

using namespace std;

class Student
//点击在此输入一行或多行代码
{
    int num;
    double score;
    char sex;
    public:
    Student(int n=0,char c='U',double s=0.0){
        num = n;
        sex = c;
        score = s;
    }
    Student(const Student &stu){
        num = stu.num;
        sex = stu.sex;
        score = stu.score;
    }
    void setScore(double s){
        score = s;
    }
    void show(){
        cout<<num<<" "<<sex<<" "<<score<<"\n";
    }
};


int main()

{
    int num;
    double score;
    char sex;
    cin>>num>>sex>>score;

    Student t1;//t1:学号0、 性别‘U’、成绩 0.0

    Student t2(num,sex,score);//t2:年龄 num岁、 性别sex、成绩score分

    Student t3(t2);// 复制构造t3，t3和当前t2一致

    t1.show();

    t2.show();

    t3.show();

    t3.setScore(70.5);//t3重新设置成绩为70.5

    t3.show();

    return 0;

}

#include<iostream>

using namespace std;



class employee //定义employee类
//点击在此输入一行或多行代码
{
    int EmpNo;
    int grade;
    int accumPay;
    public:
    employee(int e=1001,int g=1,int a=3000){
        EmpNo = e;
        grade = g;
        accumPay = a;
    }
    employee(const employee &emp){
        EmpNo = emp.EmpNo+1;
        grade = emp.grade;
        accumPay = emp.accumPay;
    }
    void promote(int g){
        grade+=g;
        accumPay += 1500*g;
    }
    void print(){
        cout<<EmpNo<<" "<<grade<<" "<<accumPay<<"\n";
    }
};


int main()

{
    int EmpNo;
    int grade;
    int accumPay;
    cin>>EmpNo>>grade>>accumPay;
    employee t1;         //t1:工号1000、 级别 1、工资 3000

    employee t2(EmpNo,grade,accumPay);//t2:工号EmpNo、 级别 grade、工资 accumPay

    employee t3(t2);// t3工号是t2工号加1，其他数据与当前t2一致

    t1.print();
    t2.print();
    t3.print();
    t3.promote(2);//t3升2级，工资共涨3000（每级涨1500）
    t3.print();

    return 0;

}
