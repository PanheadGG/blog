#include<iostream>
using namespace std;
void cal(int a[],int n,int &max_a,int &min_a)

{
    max_a=a[0];
    min_a=a[0];
    for(int i=1;i<n;i++){
        if(a[i]>max_a) max_a=a[i];
        if(a[i]<min_a) min_a=a[i];
    }
}


int main(){
	int a[10],max_a,min_a;
	cout<<"输入10个整数："<<endl;
	for(int i=0;i<10;i++) cin>>a[i];
	cal(a,10 ,max_a,min_a);
    for(int i=0;i<10;i++) cout<<a[i]<<"  ";
	cout<<endl;
	cout<<"max="<<max_a<<",min="<<min_a<<endl;
	return 0;
}

//此处添加函数实现
void Fibonacci(int f[],int n){
    if(n==1||n==2){f[n-1] = 1;return;}
    for(int i=2;i<n;i++){
        f[i]=f[i-1]+f[i-2];
    }
}

void PrintFibonacci(int f[],int n){
    for(int i=1;i<=n;i++){
        cout<<setw(8)<<f[n-i];
        if(i%5==0) cout<<"\n";
    }
}

//此处添加函数实现代码

void yh(int a[][50],int n){
    a[1][1]=1;
    if(n==1) return;
    for(int i=2;i<=n;i++){
        a[i][1]=1;
        a[i][i]=1;
        for(int j=2;j<i;j++){
            a[i][j]=a[i-1][j-1]+a[i-1][j];
        }
    }
}

void PrintYH(int a[][50],int n){
    for(int i=1;i<=n;i++){
        for(int j=1;j<=i;j++){
            cout<<a[i][j]<<" ";
        }
        cout<<"\n";
    }
}

//请添加函数实现代码
bool IsPrime(int n){
    if(n==1) return 0;
    for(int i=2;i<=sqrt(n);i++){
        if(n%i==0) return 0;
    }
    return 1;
}

int order(int a[],int b[],int n){
    int j=0;
    for(int i=0;i<n;i++){
        if(IsPrime(a[i])){
            b[j]=a[i];
            j++;
        }
    }
    return j;
}

int fun(int a[],int l)
//函数实现
{
    int max=a[0];
    int min=a[0];
    int maxAddress=0,minAddress=0,sum=0;

//写出函数实现代码
void insert(double data, double arr[], int len){
    if(data>arr[len-1]){
        arr[len]=data;
        return;
    }
    int l;
    for(int i=0;i<len;i++){
        if(data<arr[i]){
            l=i;
            break;
        }
    }
    for(int i=len;i>l;i--){
        arr[i]=arr[i-1];
    }
    arr[l]=data;
}
