/*编写一个函数，使用指针交换两个整数的值。*/
#include <iostream>
using namespace std;
void swap(int *a, int *b) 
{  //利用中间变量交换地址中的值

   int tmp=*a;
   *a=*b;
   *b=tmp;
   
}
int main() {
    int num1 = 5, num2 = 10;
    cout << "Before swap: num1 = " << num1 << ", num2 = " << num2 << endl;
    swap(&num1,&num2);     //调用函数，实参传递地址
    cout << "After swap: num1 = " << num1 << ", num2 = " << num2 << endl;
    return 0;
}

/*编写一个函数，使用指针遍历数组，计算数组中所有元素的和。*/
#include <iostream>
using namespace std;
int sumArray(int* arr, int size) {
    int sum = 0;
    for(int i=0;i<size;i++){
        sum+=arr[i];
    }
    return sum;
}
int main() {
    int arr[] = { 1, 2, 3, 4, 5 };
    int size = sizeof(arr) / sizeof(arr[0]);
    cout << "Sum of array elements: " <<   sumArray(arr,size)   << endl;
    return 0;
}

/*编写一个函数，使用指针操作字符串,在给定的字符串中查找特定字符首次出现的位置。*/
#include <iostream>
using namespace std;
int findChar(const char* str, char target) {
    int i = 0;
    while (*str != '\0') {
        if (   *str==target      ) // 依次判断指针指向的字符是否为目标字符,相等返回i的值，不相等则指针后移，下标加1
        {
            return i;
        }
        str++;
        i++;
    }
    return -1;
}
int main() {
    const char* myStr = "Hello, World!";
    char target = 'o';
    int position = findChar(myStr,target)  ;//调用函数，传递数组常量的首地址和目标字符
    if (position != -1) {
        cout << "Character '" << target << "' found at position " << position << endl;
    }
    else {
        cout << "Character not found." << endl;
    }
    return 0;
}

/*编写一个函数，返回给定数组中的最大元素，使用指针操作数组。*/
#include <iostream>
using namespace std;
int findMax(int* arr, int size) {
    int max = *arr;  //将数组的第一个值赋给max，max的类型为int
    for (int i = 1; i < size; i++) 
    { //依次比较大小，较大的值赋给max
        max = arr[i] > max ? arr[i] : max;
       
    }
    return max;
}
int main() {
    int arr[] = { 5, 9, 2, 7, 1 };
    int size = sizeof(arr) / sizeof(arr[0]);
    cout << "Max element in array: " << findMax(arr, size) << endl;
    return 0;
}

/*编写一个函数，使用指针操作数组,将一个整数数组逆序排列。*/
#include <iostream>
using namespace std;
void reverseArray(int* arr, int size) {
    int start = 0, end = size - 1;
    while (start < end) { //下标start小于end时，说明元素交换未结束，下面的语句是进行交换，并改变start和end的值
        int tmp = arr[start];
        arr[start] = arr[end];
        arr[end] = tmp;
        start++;
        end--;
        
    }
}

int main() {
    int arr[] = { 1, 2, 3, 4, 5 };
    int size = sizeof(arr) / sizeof(arr[0]);
    cout << "Original array: ";
    for (int i = 0; i < size; i++) {
        cout << arr[i] << " ";
    }
    reverseArray(arr,size); //函数调用
    cout << "\nReversed array: ";
    for (int i = 0; i < size; i++) {
        cout << arr[i] << " ";
    }
    return 0;
}

/*编写一个函数，使用指针删除字符串中的指定字符。*/
#include <iostream>
using namespace std;
void removeChar(char* str, char target) {
    // 定义两个索引变量，i用于遍历原始字符串，j用于构建新的字符串（不包含目标字符）
    int i = 0, j = 0;
    // 开始遍历原始字符串，直到遇到字符串结束符'\0'
    while  (  str[i]!='\0'  ) {
        // 如果当前字符不等于目标字符
        if  ( str[i]!=target ) {
            // 将当前字符复制到新字符串的相应位置（由j指向）
            str[j++]=str[i]  ;
        }
        // 无论当前字符是否等于目标字符，i都要向后移动一位，继续遍历下一个字符
        i++;
    }
    // 在新构建的字符串末尾添加结束符'\0'，使其成为一个合法的字符串
    str[j] = '\0';
}
int main() {
    // 定义一个字符数组并初始化，这个数组存储了一个字符串"banana"，可以对其内容进行修改
    char str[] = "banana";
    // 定义要从字符串中删除的目标字符，这里是字符'a'
    char target = 'a';
    // 输出原始字符串，展示在删除操作之前的字符串内容
    cout << "Before removal: " << str << endl;
    // 调用removeChar函数，传入字符串和目标字符，执行删除操作
    removeChar(str, target);
    // 输出删除指定字符后的字符串内容
    cout << "After removal: " << str << endl;
    return 0;
}

/*编写一个函数，使用指针统计一个字符串中单词的个数。假设单词之间以空格分隔。*/
#include <iostream>
using namespace std;
int countWords(char* str) {
    int count = 0;
    bool inWord = false;
    while   (   *str!='\0'  )//遇到\0表示结束，未遇到循环进行
    {
        // 如果当前字符不是空格且之前不在单词中，说明进入新单词
        if (*str != ' ' && !inWord) {
            inWord = true;
            count++;
        }
        // 如果当前字符是空格，说明离开单词
        else if (*str == ' ') {
            inWord = false;
        }
        str++;
    }
    return count;
}

int main() {
    char str[] = "Hello World How are you";
    cout << "Number of words: " << countWords(str) << endl;
    return 0;
}

#include <iostream>
#include <cmath>
using namespace std;

bool isPrime(int n){
    for(int i=2;i<=sqrt(n);i++)
        if(n%i==0) return 0;
    return 1;
}

bool isMirror(int n){
    return n/100==n%10;
}

int main()
{
    for(int i=100;i<=999;i++)
        if(isPrime(i)&&isMirror(i))
            cout<<i<<",";
    return 0;
}

#include <iostream>
#include <cmath>
using namespace std;
int n, q;
bool P;// 标记当前数是否为素数
int main() {
    // 遍历 100 到 200 的所有数
    for (n=100;n<=200;n++) {
       q = sqrt(n); // 计算当前数 n 的平方根,并强制类型转换为整型
       P = 1; // 标记当前数是否为素数，初始化为 true
        // 用小于等于 n 的平方根的数去尝试整除 n
        for (int i=2;i<=q;i++) {
            if (n%i == 0) {
               P = 0; // 如果能整除，说明不是素数，将标记设为 false
               break; // 一旦发现能整除，就退出循环，因为已经确定不是素数
            }
        }
        // 如果经过上面的循环，isPrime 仍然为 true，说明是素数，输出该数
        if (P == 1) {
            cout << n << ",";
        }
    }
    return 0;
}
