#include <iostream>

using namespace std;
class Circle
{
    const float PI;
    static int count;
    double r;
    double area;
public:
//点击在此输入代码    
    Circle(double r);
    void calArea();
    void showCircle();
    static int getCount(Circle c){return count;}
    friend double aveArea(Circle a,Circle b);
};
//点击在此输入一行或多行代码
int Circle::count = 0;
Circle::Circle(double r) : PI(3.14159)
{
    this->r = r;
    count++;
}
double aveArea(Circle a,Circle b){
    return (a.area + b.area) / 2;
}
void Circle::calArea(){
    area = PI * r * r;
}
void Circle::showCircle(){
    cout<<r<<" : "<<area<<"\n";
}


int main()
{
    double r1,r2;
    cin>>r1>>r2;
    Circle c1(r1),c2(r2);
    c1.calArea();
    c1.showCircle();
    c2.calArea();
    c2.showCircle();
    cout<<Circle::getCount(c1)<<endl;

    cout<<aveArea(c1,c2)<<endl;
    return 0;
}

#include <iostream>
using namespace std;

class Score
{
    int home_team;
    int opponent;
public:
    Score(int,int);
    int GetHomeTeam(){return home_team;}
    int GetOpponent(){return opponent;}
    void Display(){cout<<home_team<<" : "<<opponent<<"\n";}
    void Set(int,int);
};
Score::Score(int h,int o)
{
    home_team = h;
    opponent = o;
}
void Score::Set(int h,int o)
{
    home_team = h;
    opponent = o;
}

void input(Score score[]){
    for(int i=0;i<3;i++){
        int h,o;
        cin>>h>>o;
        score[i].Set(h,o);
    }
}

void edit(Score score[]){
    while(1){
        int i,h,o;
        cin>>i;
        if(i==0) return;
        cin>>h>>o;
        score[i-1].Set(h,o);
    }
}

void search(Score score[]){
    while(1){
        int i;
        cin>>i;
        if(i==0) return;
        score[i-1].Display();
    }
}

bool result(Score t1, Score t2, Score t3){
    int home=0,opp=0;
    t1.GetHomeTeam() > t1.GetOpponent() ? (home++) : (opp++);
    t2.GetHomeTeam() > t2.GetOpponent() ? (home++) : (opp++);
    t3.GetHomeTeam() > t3.GetOpponent() ? (home++) : (opp++);
    return home>opp;
}

int main()
{
    Score t1(98,67), t2(105,103), t3(88,96);
    Score score[] = {t1, t2, t3};
    while(1){
        cout<<"**********1.输入 3 场比赛得分**********\n";
        cout<<"**********2.修改某场比赛得分**********\n";
        cout<<"**********3.查询某场比赛得分**********\n";
        cout<<"**********4.输出冠军（主队or客队）**********\n";
        cout<<"**********0.返回主菜单**********\n";
        int i;
        cin>>i;
        switch (i)
        {
        case 1:
            input(score);
            break;
        case 2:
            edit(score);
            break;
        case 3:
            search(score);
            break;
        case 4:
            cout<<(result(score[0],score[1],score[2]) ? "主队胜！" : "客队胜！")<<"\n";
            break;
        case 0:
            return 0;
        }
    }
    return 0;
}

#include <iostream>
using namespace std;

class Boat;
class Car
{
private:
    int weight;
public:
    Car(int j){weight = j;}
    //点击在此输入代码
    friend int totalWeight(Car &aCar, Boat &aBoat);
};

class Boat
{
private:
    int weight;
public:
    Boat(int j){weight = j;}
    //点击在此输入代码
    friend int totalWeight(Car &aCar, Boat &aBoat);
};

int totalWeight(Car &aCar, Boat &aBoat)
//点击在此输入一行或多行代码
{
    return aCar.weight + aBoat.weight;
}

    cin>>w1>>w2;
