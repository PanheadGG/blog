/*定义学生类，包含姓名，和各科成绩，定义一个静态变量，累计求各科成绩之和，所有成绩之和，
输入5个学生的三科成绩，求三科各自的平均分，三科一起的平均分*/
#include <iostream>
#include <string>
using namespace std;

class Student {
private:
    string name;
    float scores[3];  // 存放三科成绩
    static float subjectTotalScores[3];  //静态数组subjectTotalScores，用于累计各科成绩总和
    static float allTotalScore;  // 静态变量allTotalScore，用于累计所有成绩总和
public:
    void setData();  // 输入学生信息（姓名和三科成绩）
    static void calculateAverages(int numStudents);  // 静态函数，用于计算各科平均分和所有成绩平均分
};

// 初始化静态数组和静态变量
float Student::subjectTotalScores[3] = {0};
float Student::allTotalScore = 0;


// 输入学生信息（姓名和三科成绩）
void Student::setData() {
   // cout << "请输入学生姓名: ";
    cin >> name;
   // cout << "请依次输入该学生的三科成绩: ";
    cin >> scores[0] >> scores[1] >> scores[2];

    // 将该学生的各科成绩累加到静态变量中
    for (int i = 0; i < 3; ++i) {
        subjectTotalScores[i] += scores[i];
    }
    allTotalScore += scores[0] + scores[1] + scores[2];
}

// 静态函数，用于计算各科平均分和所有成绩平均分
void Student::calculateAverages(int numStudents) {
    cout << "各科平均分如下: " << endl;
    for (int i = 0; i < 3; ++i) {
        cout << "第 " << (i + 1) << " 科平均分: " << subjectTotalScores[i] / numStudents << endl;
    }
    cout << "三科一起的平均分: " << allTotalScore / (numStudents * 3) << endl;
}

int main() {
    const int numStudents = 5;
    Student students[numStudents];

    // 循环输入5个学生的信息
    for (int i = 0; i < numStudents; ++i) {
        students[i].setData();
    }
//用于计算各科平均分和所有成绩平均分
    Student::calculateAverages(numStudents);

    return 0;
}

#include <iostream>
using namespace std;

class Complex
{
    double real;
    double imag;
public:
    Complex(){}
    Complex(double,double);
    Complex operator+(Complex);
    void show();
};
Complex::Complex(double r,double i)
{
    real = r;
    imag = i;
}
Complex Complex::operator+(Complex c)
{
    Complex cc(real,imag);
    cc.real = real + c.real;
    cc.imag = imag + c.imag;
    return cc;
}
void Complex::show(){
    cout<<"("<<real<<"+"<<imag<<"i)";
}

int main()
{
    Complex c1(1.1,2.2);
    Complex c2(3.3,4.4);
    Complex c3 = c1 + c2;
    c3.show();
    return 0;
}

/*定义了 “Goods” 类用于表示商品信息，通过构造函数重载实现不同方式初始化商品对象，
并借助相应成员函数在主函数中循环输出数组中各商品对象的编号、名称及价格等信息。*/
#include <iostream>
#include <string>
using namespace std;
class Goods {
private:
    string num, name;
    float price;
public:
    // 构造函数，接受一个字符串参数，用于初始化商品编号，名称为空，价格为0
        Goods(string s){
        num = s;
        name = "默认名字1";
        price = 0;
    }
    // 构造函数，接受商品编号、商品名称和商品价格三个参数，分别进行初始化
    Goods(string number, string sname, float pricetag) {
        num = number;
        name = sname;
        price = pricetag;
    }
    // 构造函数，接受一个整数参数，将其转换为浮点型后作为价格，商品编号和名称为空
    Goods(int pricetag) {
        num = "默认编号3";
        name = "默认名字3";
        price = (float)pricetag;
    }
    // 添加成员函数用于获取商品编号
    string getNum() const {
        return num;
    }
    // 添加成员函数用于获取商品名称
    string getName() const {
        return name;
    }
    // 添加成员函数用于获取商品价格
    float getPrice() const {
        return price;
    }
};
int main() {
    Goods ss[3]{ Goods("computer"), Goods("10001", "book", 35), Goods(12.5) };
    // 循环遍历数组，输出每个Goods对象的相关信息
    for (int i = 0; i < 3; ++i) {
        cout << "商品 " << (i + 1) << " 的信息如下：" << endl;
        cout << "编号：" << ss[i].getNum() << endl;  //输出对象数组元素的获取编号函数的值
        cout << "名称：" << ss[i].getName() << endl;//输出对象数组元素的获取商品名函数的值
        cout << "价格：" << ss[i].getPrice() << " 元" << endl;//输出对象数组元素的获取商品价格函数的值
        cout << endl;
    }
    return 0;
}

/*定义学生类class，类中有数据成员学号、姓名、3门课程成绩，实现学生类信息的输入、输出和求平均值*/
#include <iostream>
#include <string>
using namespace std;
class Student     //类的声明/定义
{
private:
    string name;    //姓名
    string num;     //学号
    float score[3];   //三门课程成绩
public:
    void setdate();    //输入学生信息
    float stu_ave();    //三门课平均成绩
    void print();       //显示学生数据信息   
};
//类的实现
void Student::setdate()          //类的成员函数setdate()的实现
{ cout << "请输入学生的学号：";
    cin >> num;
    cout << "请输入学生的姓名：";
    cin >> name;
    cout << "请输入学生三门课程的成绩：";
    cin >> score[0] >> score[1] >> score[2];  }
float Student::stu_ave()
{ return (score[0] + score[1] + score[2]) / 3;  }
void Student::print()
{ cout << "学生的学号：";
    cout << num << endl;
    cout << "学生的姓名：";
    cout << name << endl;
    cout << "学生三门课程的成绩：";
    cout << score[0] << "   " << score[1] << "   " << score[2] << endl;  }
int main()
{   Student s; // 创建一个Student类的对象s，用于存储和操作单个学生的相关信息
    s.setdate();  // 对象s调用Student类中的setdate函数，通过用户输入来设置该学生对象s的学号、姓名以及三门课程成绩等信息
    // 调用Student类中的stu_ave函数来计算该学生三门课程成绩的平均值，并将结果输出显示
    cout << "该学生三门课程的平均成绩为：" << s.stu_ave() << endl;  
    s.print();  // 调用Student类中的print函数，将该学生对象s的学号、姓名以及三门课程成绩等信息输出显示
    return 0;  // 表示程序正常结束，向操作系统返回值0
}
