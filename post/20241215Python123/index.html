class DataOperation
{
public:
    DataOperation(){}
    ~DataOperation(){}
    void sort(double arr[],int size);
    double findMaxValue(double arr[],int size);
    double findMinValue(double arr[],int size);
    double ave(double arr[],int size);
    void sortObject(Student arr[], int size, int sortflag);
};
void DataOperation::sort(double arr[],int size){
    std::sort(arr,arr+size,cmp);
}
double DataOperation::findMaxValue(double arr[],int size){
    double max = arr[0];
    for(int i=1;i<size;i++)
        max = max > arr[i] ? max : arr[i];
    return max;
}
double DataOperation::findMinValue(double arr[],int size){
    double min = arr[0];
    for(int i=1;i<size;i++)
        min = min < arr[i] ? min : arr[i];
    return min;
}
double DataOperation::ave(double arr[],int size){
    double sum = 0;
    for(int i=0;i<size;i++) sum += arr[i];
    return sum / size;
}
void DataOperation::sortObject(Student arr[], int size, int sortflag){
    for(int i=0;i<size-1;i++){
        for(int j=0;j<size-1;j++){
            switch (sortflag)
            {
            case 1:
                if(arr[i].getName().compare(arr[i+1].getName())==-1){
                    Student tmp = arr[i];
                    arr[i] = arr[i+1];
                    arr[i+1] = tmp;
                }
                break;
            case 2:
                if(arr[i].getNum()>arr[i+1].getNum()){
                    Student tmp = arr[i];
                    arr[i] = arr[i+1];
                    arr[i+1] = tmp;
                }
                break;
            case 3:
                if(arr[i].getAveScore()<arr[i+1].getAveScore()){
                    Student tmp = arr[i];
                    arr[i] = arr[i+1];
                    arr[i+1] = tmp;
                }
                break;
            }
        }
    }
}

#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

class Person {
protected:
    string name;
    char sex;
    int age;

public:
    Person(string n, char s, int a) : name(n), sex(s), age(a) {}

    virtual void eat() {
        cout << name << " is eating." << endl;
    }

    virtual void sleep() {
        cout << name << " is sleeping." << endl;
    }

    virtual void work() {
        cout << name << " is working." << endl;
    }

    virtual void printInfo() {
        cout << "Name: " << name << ", Sex: " << sex << ", Age: " << age << endl;
    }

    string getName() const {
        return name;
    }
};

class Student : public Person {
private:
    string school;
    string s_number;
    float totalScore;
    float aveScore;
    float scores[3]; // Assuming 3 subjects: Chinese, Math, English

public:
    Student(string n, char s, int a, string sch, string sn, float sc[])
        : Person(n, s, a), school(sch), s_number(sn) {
        totalScore = 0;
        for (int i = 0; i < 3; ++i) {
            scores[i] = sc[i];
            totalScore += sc[i];
        }
        aveScore = totalScore / 3;
    }

    void work() override {
        cout << name << " is studying." << endl;
    }

    void printInfo() override {
        Person::printInfo();
        cout << "School: " << school << ", Student Number: " << s_number
            << ", Total Score: " << totalScore << ", Average Score: " << aveScore << endl;
    }

    string getStudentNumber() const {
        return s_number;
    }

    float getAverageScore() const {
        return aveScore;
    }
};

class DataOperation {
public:
    static void sortObject(vector<unique_ptr<Student>>& arr, int sortflag) {
        switch (sortflag) {
        case 0: // Sort by name
            sort(arr.begin(), arr.end(), [](const unique_ptr<Student>& a, const unique_ptr<Student>& b) {
                return a->getName() < b->getName();
                });
            break;
        case 1: // Sort by student number
            sort(arr.begin(), arr.end(), [](const unique_ptr<Student>& a, const unique_ptr<Student>& b) {
                return a->getStudentNumber() < b->getStudentNumber();
                });
            break;
        case 2: // Sort by average score
            sort(arr.begin(), arr.end(), [](const unique_ptr<Student>& a, const unique_ptr<Student>& b) {
                return a->getAverageScore() > b->getAverageScore();
                });
            break;
        }
    }
};

int main() {
    float scores1[] = { 85, 90, 78 };
    float scores2[] = { 92, 88, 85 };

    vector<unique_ptr<Student>> students = {
        make_unique<Student>("Alice", 'F', 20, "XYZ University", "S001", scores1),
        make_unique<Student>("Bob", 'M', 21, "XYZ University", "S002", scores2)
    };

    cout << "Before sorting:" << endl;
    for (const auto& student : students) {
        student->printInfo();
    }

    DataOperation::sortObject(students, 2); // Sort by average score

    cout << "\nAfter sorting by average score:" << endl;
    for (const auto& student : students) {
        student->printInfo();
    }
    return 0;
}
