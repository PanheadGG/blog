class Student {
    private String name;
    private double score;
    private static int num = 0;

    public Student() {
        num++;
    }

    public Student(String name, double score) {
        this.name = name;
        this.score = score;
        num++;
    }

    public String getName() {
        return this.name;
    }

    public double getScore() {
        return this.score;
    }

    public void setName(String name) {
        this.name = name;
    }

    public void setScore(double score) {
        this.score = score;
    }

    public static int getNum(){
        return num;
    }
}

public class Test {
    public static void main(String[] args){
        Student stu1 = new Student();
        stu1.setName("Li Hua");
        stu1.setScore(66.6);
        System.out.printf("Student name %s,score %f\n",stu1.getName(),stu1.getScore());
        Student stu2 = new Student("Zhang San",99.9);
        System.out.printf("Student name %s,score %f\n",stu2.getName(),stu2.getScore());

        System.out.print("Num of Student: ");
        System.out.println(Student.getNum());
    }
}

public class YangHuiSanJiao {
    public static void main(String[] args){
        int[][] yhsj = new int[8][8];
        for(int i=0;i<8;i++){
            for(int j=0;j<=i;j++){
                if(j==0||j==i){
                    yhsj[i][j] = 1;
                } else {
                    yhsj[i][j] = yhsj[i-1][j-1] + yhsj[i-1][j];
                }
            }
        }
        for(int i=0;i<8;i++){
            for(int j=0;j<=i;j++){
                System.out.print(yhsj[i][j]+"\t");
            }
            System.out.println();
        }

        //下面等腰三角形打印参考https://www.cnblogs.com/JumperMan/p/6759422.html
        for(int i =0;i<8;i++) {
            //打印空格字符串
            System.out.printf("%"+(8-i)*2+"s","");
            for(int j=0;j<=i;j++) {
                System.out.printf("%4d",yhsj[i][j]);
            }
            System.out.println();
        }
    }
}

        for (int i = 1; i <= 9; i++) {
            for (int j = 1; j <=i; j++) {
                System.out.printf("%d*%d=%d\t",i,j,i*j);

public class Work4 {
    public static int jieCheng(int n){
        if(n==0||n==1) return 1;
        return n*jieCheng(n-1);
    }

    public static void main(String[] args) {
        int result = 1;
        for (int i = 2; i <= 10; i++) {
            result *= i;
        }
        System.out.println("使用循环: "+result);
        System.out.println("使用递归: "+jieCheng(10));
    }
}

import java.util.Scanner;

public class Fibonacci {
    public static int fibonacci(int n) {
        if (n == 1 || n == 2) return 1;
        return fibonacci(n - 1) + fibonacci(n - 2);
    }

    public static void main(String[] args) {
        int month = 1;
        System.out.print("递归实现斐波那契数列，请输入n的值: ");
        Scanner scanner = new Scanner(System.in);
        if (scanner.hasNextInt()) {
            month = scanner.nextInt();
        }
        System.out.println(fibonacci(month));

        //数组实现斐波那契数列
        int[] fib = new int[20];
        fib[0] = 1;
        fib[1] = 1;
        for (int i = 2; i < 20; i++) {
            fib[i] = fib[i - 1] + fib[i - 2];
        }
        System.out.println("数组实现斐波那契数列: ");
        for (int i = 0; i < 20; i++) {
            System.out.print(fib[i]+" ");
        }
    }
}
