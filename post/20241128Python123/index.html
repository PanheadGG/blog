    cin>>orderflag;

    for(i = 0; i <n; i++)
         cout<<setw(5)<<a[i];

void sorting(int x[], int n,char orderFlag)
{
    if(orderFlag=='A'){
        for(int i=0;i<n-1;i++){
            for(int j=0;j<n-1-i;j++){
                if(x[j]>x[j+1]){
                    int tmp = x[j];
                    x[j] = x[j+1];
                    x[j+1] = tmp;
                }
            }
        }
    }
     if(orderFlag=='D'){
        for(int i=0;i<n-1;i++){
            for(int j=0;j<n-1-i;j++){
                if(x[j]<x[j+1]){
                    int tmp = x[j];
                    x[j] = x[j+1];
                    x[j+1] = tmp;
                }
            }
        }
    }
}

#include <iostream>
using namespace std;
int a[100];

int main()
{
    int n;
    double sum;
    cin>>n;
    int max = -1e8+1;
    int min = 1e8-1;
    for(int i=0;i<n;i++) {
        cin>>a[i];
        max = a[i]>max?a[i]:max;
        min = a[i]<min?a[i]:min;
        sum+=(float)a[i]; 
    }
    cout<<"max="<<max<<"\n";
    cout<<"min="<<min<<"\n";
    cout<<"ave="<<sum/n;

    return 0;
}

#include <iostream>
#include <iomanip>
#include <cstring>
using namespace std;
int a[100],hashMap[99999];

int main()
{
    memset(hashMap,0,sizeof(hashMap));
    int n;
    cin>>n;
    for(int i=0;i<n;i++){
        cin>>a[i];
        hashMap[a[i]]++;
        }
    cout<<"原来数组元素\n";
    for(int i=0;i<n;i++) cout<<setw(5)<<a[i];
    cout<<"\n新的数组元素: ";
    for(int i=0;i<n;i++){
        if(hashMap[a[i]]!=0){
            cout<<setw(5)<<a[i];
            hashMap[a[i]]=0;
        }
    }

    return 0;
}
