#include <iostream>
using namespace std;

int gcd(int a,int b)
{
    int max=1;
    for(int i=1;i<(a<b?a:b);i++){
        if(a%i==0&&b%i==0) max=i;
    }
    return max;
}
int main()
{
    int m,n;
    cin>>m>>n;
    cout << gcd(m,n) << endl;
    return 0;
}

#include <iostream>
using namespace std;

int gcd(int a, int b)
{
    if (a % b == 0)
        return b;
    else
        return gcd(b, a % b);
}

int main()
{
    int a, b;
    cin >> a >> b;
    cout << gcd(a, b);
    return 0;
}

#include <iostream>
using namespace std;

long fac(int n)//计算数n的阶乘
{
    if(n==0||n==1) return 1;
    int result=1;
    for(int i=2;i<=n;i++) result*=i;
    return result;
}

long sum1(int n)//计算1!+2!+...n!
{
    int sum=0;
    for(int i=1;i<=n;i++) sum+=fac(i);
    return sum;
}

long sum2(int n)//计算1!+(1!+2!)+(1!+2!+3!)+...(1!+2!+...n!)
{
    int sum=0;
    for(int i=1;i<=n;i++) sum+=sum1(i);
    return sum;
}

int main()
{
    int n;
    cin>>n;
    cout << sum2(n)<< endl;
    return 0;
}

//编写重载函数可分别求取两个整数，三个整數，两个双精度数，三个双精度数的最大值。并在主函数中分别测试。
#include <iostream>
using namespace std;

int max(int a,int b){
    return a>b?a:b;
}

double max(double a,double b){
    return a>b?a:b;
}

int max(int a,int b,int c){
    int max=a;
    if(b>max) max=b;
    if(c>max) max=c;
    return max;
}

double max(double a,double b,double c){
    double max=a;
    if(b>max) max=b;
    if(c>max) max=c;
    return max;
}

int main()
{
    int a,b,c;
    double d,e,f;
    cin>>a>>b>>c>>d>>e>>f;
    cout<<max(a,b)<<" "<<max(a,b,c)<<" "<<max(d,e)<<" "<<max(d,e,f);
    return 0;
}

#include <iostream>
using namespace std;

void print(int n,char ch){
    for(int i=1;i<=n;i++){
        for(int j=1;j<=n-i;j++){
            cout<<" ";
        }
        for(int k=1;k<=2*i-1;k++){
            cout<<ch;
        }
        cout<<"\n";
    }
}

int main()
{
    int n;
    char ch;
    cin>>n>>ch;
    print(n,ch);
    return 0;
}

#include <iostream>
using namespace std;
//汉诺塔递推表达式
//f(n)=2f(n-1)+1且f(1)=1
int hanoiTower(int n){
    if(n==1) return 1;
    return 2*hanoiTower(n-1)+1;
}

int main()
{
    int n;
    cin>>n;
    cout<<hanoiTower(n);
    return 0;
}

#include <iostream>
using namespace std;

bool isPrime(int n){
    if (n <= 1) return false;
    for(int i = 2; i * i <= n; i++){
        if(n % i == 0) return false;
    }
    return true;
}

int main()
{
    int n;
    cin>>n;
    cout<<n<<(isPrime(n)?" is a prime!\n":" is not a prime!\n");
    for (int i = 2; i <= n / 2; i++) {
        if (isPrime(i)) {
            int j = n - i;
            if (isPrime(j)) {
                cout << n << "=" << i << "+" << j << endl;
            }
        }
    }
    return 0;
}
