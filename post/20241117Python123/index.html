#include <iostream>
#include <cmath>
using namespace std;
bool symm(long n);

int main()
{

    long m;
    int a, b;
    cin >> a >> b;

    for (m = a; m < b; m++)
        if (symm(m) && symm(m * m) && symm(m * m * m))
            cout << "m=" << m << "  m*m=" << m * m << "  m*m*m=" << m * m * m << endl;
    return 0;
}

// 此处添加函数的实现代码

bool symm(long n)
{
    long a = n;
    long re = 0;
    while (n > 0)
    {
        re = re * 10 + n % 10;
        n /= 10;
    }
    return a == re;
}

    double k, r, s;
    cin >> r >> s;

    if (r * r <= s * s)
        k = sqrt(tsin(r) * tsin(r) + tsin(s) * tsin(s));
    else
        k = tsin(r * s) / 2;

    cout << k << endl;
    return 0;
}

double tsin(double x) {
    double sum = 0;
    double term = x;
    double n = 0;
    while (fabs(term) >= 1e-6) {
        sum += term;
        n++;
        term *= (-1) * x * x / ((2 * n) * (2 * n + 1));
    }
    // return sin(x);
    return sum;
}

#include <iostream>
#include <cmath>
using namespace std;

int checkNumber(int n){
    int a=1;

    while(1){
        n/=10;
        if(n!=0) a++;
        else break;
    }
    return a;
}

int isRose(int n){
    int a,b,c,d;
    a=n/1000;
    b=n/100%10;
    c=n/10%10;
    d=n%10;
    if(pow(a,4)+pow(b,4)+pow(c,4)+pow(d,4)==n) return true;
    else return false;
}

int main()
{
    int a,b,n=0;
    cin>>a>>b;
    if(checkNumber(a)!=4) cout<<a<<"不是四位数！"<<"\n";
    if(checkNumber(b)!=4) cout<<b<<"不是四位数！"<<"\n";
    if(checkNumber(a)!=4||checkNumber(b)!=4) return 0;
    for(int i=a;i<=b;i++){
        if(isRose(i)){cout<<i<<"\n";n++;}
    }
    if(n==0) cout<<"此区间没有四位玫瑰数";

    return 0;
}

#include<iostream>
using namespace std;

bool isPrime(int n){
    if (n <= 1) return false;
    for(int i = 2; i * i <= n; i++){
        if(n % i == 0) return false;
    }
    return true;
}

int main() {
    int a,n,num=0;
    cin >> a >> n;

    for(int i=a;i<=n;i++){
        if(isPrime(i)){
            cout<<i;
            num++;
            if(num==10){
                num=0;
                cout<<"\n";
                continue;
            }
            cout<<",";
        }
    }

    cout<<"\n";
    for (int z = a; z <= n; z++) {
        if (z % 2 == 0) {
            for (int i = 2; i <= z / 2; i++) {
                if (isPrime(i) && isPrime(z - i)) {
                    cout << z << "=" << i << "+" << (z - i) << endl;
                    break;
                }
            }
        }
    }

    return 0;
}

#include <iostream>

double fact(int n);
double c(int m, int n, int k);

int main()
{
    int m, n, k;
    cin >> m >> n;
    k = m - n;
    cout << c(m, n, k);
    return 0;
}

// 请写出fact函数和c函数的实现
double fact(int n)
{
    double result = 1;
    for (int i = 1; i <= n; i++)
    {
        result *= i;
    }
    return result;
}

double c(int m, int n, int k)
{
    return fact(m) * fact(n) / fact(m - n);
}

#include <iostream>
#include <cmath>
using namespace std;

// 判断一个数是否为素数的函数
bool IsPrime(long num)
{
    if (num < 2)
        return false;
    for (int i = 2; i <= sqrt(num); i++)
    {
        if (num % i == 0)
            return false;
    }
    return true;
}

// 判断一个数是否为超级素数的函数
bool IsSuperPrime(long num)
{
    long temp = num;
    int digit_count = 0;
    // 计算数字的位数
    while (temp > 0)
    {
        digit_count++;
        temp /= 10;
    }
    temp = num;
    for (int i = 0; i < digit_count; i++)
    {
        if (!IsPrime(temp))
            return false;
        temp /= 10;
    }
    return true;
}

int main()
{
    long i, j, k;
    int a, b, n = 0;
    cin >> a >> b;
    for (i = a; i <= b; i = i + 1)
    {
        if (IsSuperPrime(i))
        {
            n++;
            if (n % 5 != 0)
                cout << i << ",";
            else
                cout << i << endl;
        }
    }
}
