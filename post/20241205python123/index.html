第一题

#include <iostream>
using namespace std;

bool isBlue(int a, int b){
   return a<b;
}

const int N=9;
int a[]={1,3,5,7,9,11,13,15,17};

int main()
{
    int l=-1,r=N,mid,numToSearch=7;
    while(l+1!=r){
        mid=(l+r)/2;
        if(isBlue(a[mid],numToSearch)) l=mid;
        else r=mid;
    }
    cout<<"查找成功,其位置是(第一个为0开始计数):"<<mid;
    return 0;
}

第三题

#include <iostream>
#include <iomanip>
using namespace std;

double score[100][100];

int main()
{
    int n,m;
    cin>>n>>m;
    for(int i=0;i<n;i++){
        double sum=0;
        int max=0,min=100;
        for(int j=0;j<m;j++){
            cin>>score[i][j];
            max=score[i][j]>max?score[i][j]:max;
            min=score[i][j]<min?score[i][j]:min;
            sum+=score[i][j];
        }
        score[i][m]=(sum-max-min)/(m-2);
    }
    cout<<"每个选手的分数和平均得分如下："<<"\n";
    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++){
            cout<<" "<<fixed<<setprecision(2)<<score[i][j];
        }
        cout<<" "<<score[i][m]<<"\n";
    }
    return 0;
}

/*一场教学比赛，有num_players名参赛选手，num_judges个评委为其打分。选手的最后得分是去掉一个最高分和一个最低分后求到的平均分(小数点后保留2位)，请存储所有评分以及每个选手最后得分。*/
#include <iostream>
#include <iomanip>
using namespace std;
int main() {
    int num_players, num_judges; // num_players表示选手数量, num_judges为评委数量
    //cout << "请输入选手数量: "; 
    cin>>num_players;; 
  //  cout << "请输入评委数量: ";  
    cin>>num_judges;;
    // 二维数组存储评委打分，第一维表示选手，第二维表示评委
    double scores[100][100];  // 假设最多100个选手和100个评委
    // 输入每个选手的分数
    for (int i = 0; i < num_players; i++) {
   //     cout << "请输入第 " << (i + 1) << " 个选手的 " << num_judges << " 个分数：" << endl;
        double sum = 0;
        for (int j = 0; j < num_judges; j++) 
        {
            cin>>scores[i][j];  //依次输入评委的打分
            sum+=scores[i][j];  //得分累计求和
            }
           scores[i][99]=scores[i][0]  ;  //假定最高分为第一个评委的分数
           scores[i][98]=scores[i][0]   ;  //假定最低分为第一个评委的分数
        // 比较大小，得到每个选手的最低分和最高分
        for (   int j = 0; j < num_judges; j++   ) {
            if(scores[i][j]>scores[i][99])
                scores[i][99]=scores[i][j];
            if(scores[i][j]<scores[i][98])
                scores[i][98]=scores[i][j];
        }
       scores[i][num_judges] = (sum-scores[i][99]-scores[i][98])/(num_judges-2);     // 计算平均得分并存入最后一列
    }
    // 输出每个选手的分数和平均得分
    cout << "每个选手的分数和平均得分如下：" << endl;
    for (int i = 0; i < num_players; i++) {
        for (int j = 0; j < num_judges + 1; j++) {
            cout << fixed << setprecision(2) << setw(6) << scores[i][j];    
        }
        cout << endl;  
    }
    return 0;
}

第八题

#include <iostream>
#include <iomanip>
using namespace std;

int a[10][10];

int main()
{
    int m, n, count = 0;
    cin >> m >> n;
    int u = 0, b = m - 1, l = 0, r = n - 1;
    while (count != m * n) {
        for(int i=u;i<=b;i++){
            a[i][r] = ++count;
        }
        r--;
        for(int i=r;i>=l;i--){
            a[b][i] = ++count;
        }
        b--;
        for(int i=b;i>=u;i--){
            a[i][l] = ++count;
        }
        l++;
        for(int i=l;i<=r;i++){
            a[u][i] = ++count;
        }
        u++;
    }
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            cout << a[i][j] << "\t";
        }
        cout << "\n";
    }

    return 0;
}

// 递归计算整数各位数字之和
#include <iostream>
using namespace std;
int sumOfDigits(int n)  //以下代码实现函数递归
{
    int sum;
    while(n){
        sum+=n%10;
        n/=10;
    }
return sum;
}
int main() {
